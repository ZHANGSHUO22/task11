library(MASS)
library(SMSdata)
data(carc)

lda1<- lda(C ~ T+D+G+L+W+Tr+R+H+M+P+R77+R78,data=carc)
pred<- predict(lda1)
ldahist(data = pred$x[,1], g=carc$C)

require(nnet)
## Loading required package: nnet
par(mfrow=c(2,2))
mowers.gr = expand.grid(seq(min(carc$T),max(carc$T),len=1000),seq(min(carc$G),max(carc$G),len=1000))
names(mowers.gr) = c("T","G")
mowe.ii = unique(mowers.gr$T)
mowe.ll = unique(mowers.gr$G)
attach(carc)
## The following object is masked from package:base:
## 
##     T
for (k in 1:4) {
  mow.nn = nnet( factor(C)~ T+G,size=50, data=carc)
  plot(T,G,pch=21, bg = c("red","blue"), main = "Generalized Linear Model")
  contour(mowe.ii,mowe.ll,
          matrix(as.numeric(as.factor(predict(mow.nn,new=mowers.gr,type="class"))),1000,1000),
          nlev=1,drawlabels=FALSE,add=T)
}
